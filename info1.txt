fname=input('Enter the name of the file :')
handle=open(fname)
bars={this is an empty dictionary}

# iterating through the list

for line in handle :
    space=line.rstrip()
    words=line.split()
    for word in words :
        bars[word]=bars.get(word,0)+1
#print(bars)
# Bigkey is an object

bigkey=None
bigvalue=None
for key,value in bars.items() :
    if bigvalue is None or value > bigvalue :
        bigkey=key
        bigvalue=value
print(bigkey,bigvalue)
